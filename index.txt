function handleFile(file){
  if(!file)return;currentFile=file;
  const reader=new FileReader();
  reader.onload=e=>{
    const img=document.getElementById('previewImg');
    img.src=e.target.result;img.style.display='block';
    document.getElementById('previewPlaceholder').style.display='none';
  };
  reader.readAsDataURL(file);
  document.getElementById('analyseBtn').disabled=false;
  document.getElementById('dropZone').querySelector('.drop-label').textContent=file.name;
}

async function analyseSelfie(){
  if(!currentFile)return;
  setLoading('analyseBtn',true);hideError('selfieError');showLoading();
  const fd=new FormData();fd.append('file',currentFile);fd.append('top_n',8);
  try{
    const res=await fetch(API+'/analyse',{method:'POST',body:fd});
    const data=await res.json();
    if(!res.ok)throw new Error(data.detail||'Analysis failed');
    renderAll(data);
  }catch(e){showError('selfieError',e.message);hideLoading();}
  finally{setLoading('analyseBtn',false);}
}

function labToRgb(L,A,B){
  let fy=(L+16)/116,fx=A/500+fy,fz=fy-B/200;
  const d=6/29;
  const x=0.95047*(fx>d?Math.pow(fx,3):3*d*d*(fx-4/29));
  const y=1.*(fy>d?Math.pow(fy,3):3*d*d*(fy-4/29));
  const z=1.08883*(fz>d?Math.pow(fz,3):3*d*d*(fz-4/29));
  let r=x*3.2406+y*-1.5372+z*-0.4986;
  let g=x*-0.9689+y*1.8758+z*0.0415;
  let b=x*0.0557+y*-0.204+z*1.057;
  const gm=c=>c>0.0031308?1.055*Math.pow(c,1/2.4)-0.055:12.92*c;
  r=Math.round(Math.min(255,Math.max(0,gm(r)*255)));
  g=Math.round(Math.min(255,Math.max(0,gm(g)*255)));
  b=Math.round(Math.min(255,Math.max(0,gm(b)*255)));
  return 'rgb('+r+','+g+','+b+')';
}
